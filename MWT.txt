LibraryContext.js

import React, { createContext, useState } from "react";

export const LibraryContext = createContext();

export const LibraryProvider = ({ children }) => {
  const [books, setBooks] = useState([
    { id: 1, title: "Atomic Habits", author: "James Clear", read: false },
    { id: 2, title: "The Alchemist", author: "Paulo Coelho", read: false },
    { id: 3, title: "Rich Dad Poor Dad", author: "Robert Kiyosaki", read: false },
  ]);

  const toggleReadStatus = (id) => {
    setBooks((prevBooks) =>
      prevBooks.map((book) =>
        book.id === id ? { ...book, read: !book.read } : book
      )
    );
  };

  return (
    <LibraryContext.Provider value={{ books, toggleReadStatus }}>
      {children}
    </LibraryContext.Provider>
  );
};

import React, { useContext, useEffect, useRef, useState } from "react";
import { LibraryContext } from "../context/LibraryContext";
import BookCard from "../components/BookCard";

const Library = () => {
  const { books, toggleReadStatus } = useContext(LibraryContext);
  const [search, setSearch] = useState("");
  const searchInputRef = useRef(null);

  // ‚úÖ Focus input when page loads
  useEffect(() => {
    searchInputRef.current.focus();
  }, []);

  // ‚úÖ Alert when books change
  useEffect(() => {
    alert("Library Updated!");
  }, [books]);

  const filteredBooks = books.filter((book) =>
    book.title.toLowerCase().includes(search.toLowerCase())
  );

  return (
    <div style={{ padding: "20px" }}>
      <h2>üìñ Library</h2>

      <input
        type="text"
        placeholder="Search Books..."
        ref={searchInputRef}
        value={search}
        onChange={(e) => setSearch(e.target.value)}
        style={{ marginBottom: "20px", padding: "5px" }}
      />

      {filteredBooks.map((book) => (
        <BookCard
          key={book.id}
          book={book}
          toggleReadStatus={toggleReadStatus}
        />
      ))}
    </div>
  );
};

export default Library;

RegistrationPage.js:
import React, { useContext, useState, useEffect, useRef } from "react";
import { StudentContext } from "../context/StudentContext";

function RegistrationPage() {
  const { student } = useContext(StudentContext);

  const [workshop, setWorkshop] = useState("");
  const [success, setSuccess] = useState(false);

  const formRef = useRef();

  const handleSubmit = (e) => {
    e.preventDefault();
    setSuccess(true);
  };
  


  // success message disappear after 3 sec
  useEffect(() => {
    if (success) {
      const timer = setTimeout(() => {
        setSuccess(false);
      }, 3000);

      return () => clearTimeout(timer);
    }
  }, [success]);

  const handleReset = () => {
    setWorkshop("");
    formRef.current.reset();
  };

  return (
    <div>
      <h2>Workshop Registration</h2>

      <p>Name: {student.name}</p>
      <p>Roll No: {student.rollNo}</p>

      <form ref={formRef} onSubmit={handleSubmit}>

        <select
          value={workshop}
          onChange={(e) => setWorkshop(e.target.value)}
        >
          <option value="">Select Workshop</option>
          <option value="AI">AI Workshop</option>
          <option value="Web">Web Development</option>
          <option value="Cloud">Cloud Computing</option>
        </select>

        <br /><br />

        <button type="submit">Register</button>
        <button type="button" onClick={handleReset}>
          Reset
        </button>

      </form>

      {success && <p>Registration Successful</p>}
    </div>
  );
}

export default RegistrationPage;

CartContext:
import React, { createContext, useState } from "react";

export const CartContext = createContext();

export const CartProvider = ({ children }) => {
  const [cartItems, setCartItems] = useState([]);

  const addToCart = (product) => {
    setCartItems((prev) => {
      const exists = prev.find(p => p.id === product.id);
      if (exists) {
        return prev.map(p =>
          p.id === product.id ? { ...p, qty: p.qty + 1 } : p
        );
      }
      return [...prev, { ...product, qty: 1 }];
    });
  };

  return (
    <CartContext.Provider value={{ cartItems, setCartItems, addToCart }}>
      {children}
    </CartContext.Provider>
  );
};

Cart.js:
import React, { useContext, useState, useEffect, useRef } from "react";
import { CartContext } from "../context/CartContext";

function Cart() {
  const { cartItems, setCartItems } = useContext(CartContext);
  const [total, setTotal] = useState(0);
  const prevTotal = useRef(0);

  const updateQty = (id, change) => {
    setCartItems(prev =>
      prev.map(item =>
        item.id === id
          ? { ...item, qty: Math.max(1, item.qty + change) }
          : item
      )
    );
  };

  useEffect(() => {
    const newTotal = cartItems.reduce(
      (sum, item) => sum + item.price * item.qty,
      0
    );

    prevTotal.current = total;
    setTotal(newTotal);
  }, [cartItems]);

  return (
    <div>
      <h2>Cart</h2>

      {cartItems.map(item => (
        <div key={item.id}>
          {item.name} - ‚Çπ{item.price}
          <button onClick={() => updateQty(item.id, -1)}>-</button>
          {item.qty}
          <button onClick={() => updateQty(item.id, 1)}>+</button>
        </div>
      ))}

      <h3>Total: ‚Çπ{total}</h3>

      {total > prevTotal.current && <p>Total Increased üìà</p>}
      {total < prevTotal.current && <p>Total Decreased üìâ</p>}
    </div>
  );
}

export default Cart;
